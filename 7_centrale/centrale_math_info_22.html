def temps(a, b):
    Sn = 0
    T = 0
    while -a <= Sn <= b:
        T += 1
        if rd.random() < 0.5:
            Sn += 1
        else:
            Sn -= 1
    return T

    m = 0

import scipy.optimize as resol

def fact(n):
    if n == 0:
        return 1
    else:
        return n*fact(n-1)

def f(n):
    def aux(x):
        s = 0
        for k in range(n):
            s += float(x**k/fact(k))*np.exp(-x)

def D1(n, a, b):
    if n == 2:
        return a**2 -2*b
    if n == 3:
        return a**3 - 3*a*b
    return a*D1(n-1, a, b) - b*D1(n-2, a, b)

def A(n):
    M = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if abs(i-j) == 1:
                M[i][j] = 1
    return M

def P(n):
    M = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            M[i][j] = np.sin((i + 1)*(j + 1)*np.pi/(n+1))
    return M

def B(n):
    p = P(n)
    return np.linalg.inv(p).dot(A(n)).dot(p)
