Attracteurs#

Calcul des attracteurs avec la fonction du cours :

def attracteurs(G, fA):
    d = {} # d[v] = True si v est un attracteur
    def aux(v): # dÃ©termine si v est un attracteur
        if v not in d:
            succ = [aux(w) for w in G[v]]
            if len(G[v]) == 0:
                d[v] = not fA(v)
            elif fA(v):
                d[v] = any(succ)
            else:
                d[v] = all(succ)
        return d[v]
    return [v for v in G if aux(v)]

Autre version possible :

from functools import cache

def attracteurs(G, fA):
    @cache
    def aux(v):
        succ = map(aux, G[v])
        if G[v] == []:
            return not fA(v)
        if fA(v):
            return any(succ)
        return all(succ)
    return filter(aux, G)

Jeu de Nim#

def nim(n):
    G = {}
    for i in range(1, n + 1):
        for j in ['A', 'B']:
            if (i, j) not in G:
                G[(i, j)] = []
            for k in range(i + 1, i + 4):
                if k <= n:
                    G[(i, j)].append((k, 'A' if j == 'B' else 'B'))
    return G
