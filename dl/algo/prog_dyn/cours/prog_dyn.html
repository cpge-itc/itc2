def knapsack(c, w, v):
    """ 
    Renvoie la valeur maximum que l'on peut mettre
    dans un sac √† dos de capacit√© c.
    Le i√®me objet a pour poids w[i] et valeur v[i].
    """
    n = len(w) # nombre d'objets
    dp = [[0]*(n + 1) for i in range(c + 1)]
    # dp[i][j] = valeur max dans un sac de capacit√© i
    # o√π j est le nombre d'objets autoris√©s
    for i in range(1, c + 1):
        for j in range(1, n + 1):
            if w[j - 1] <= i:
                x = v[j - 1] + dp[i - w[j - 1]][j - 1]
                dp[i][j] = max(dp[i][j - 1], x)
            else:
                dp[i][j] = dp[i][j - 1]
    return dp[c][n]

def bellman(g, r):
    n = len(g)
    dist = [float("inf") for i in range(n)]
    dist[r] = 0
    for k in range(n - 2):
        for u in range(n):
            for v in range(len(g[u])):
                if g[u][v] != 0:
                    dist[v] = min(dist[v], dist[u] + g[u][v])
    return dist

import copy

def floydwarshall(g):
    n = len(g)
    dist = copy.deepcopy(g)
    for k in range(n):
        for u in range(n):
            for v in range(n):
                dist[u][v] = min(dist[u][v], dist[u][k] + dist[k][v])
    return dist
